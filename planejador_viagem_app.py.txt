# === Bibliotecas ===
import streamlit as st
from crewai import Agent, Task, Crew
from langchain.chat_models import ChatOpenAI
import folium
from streamlit_folium import st_folium
import requests

# === Fun√ß√£o para obter coordenadas ===
def obter_coordenadas(cidade):
    url = f"https://nominatim.openstreetmap.org/search?format=json&q={cidade}"
    resposta = requests.get(url).json()
    if resposta:
        return float(resposta[0]['lat']), float(resposta[0]['lon'])
    return None, None

# === Fun√ß√£o principal ===
def planejar_viagem(destino, dias, interesses):
    llm = ChatOpenAI(model="gpt-4", temperature=0.5)

    planejador = Agent(
        role="Planejador de Viagem",
        goal=f"Criar um roteiro personalizado de {dias} dias para {destino}, considerando interesses como {interesses}",
        backstory="Especialista em criar roteiros personalizados para qualquer tipo de viajante.",
        verbose=True,
        llm=llm
    )

    pesquisador = Agent(
        role="Pesquisador de Destinos",
        goal=f"Pesquisar atra√ß√µes e dicas √∫teis sobre {destino}, focando em {interesses}",
        backstory="Especialista em buscar dados atualizados sobre turismo e experi√™ncias locais.",
        verbose=True,
        llm=llm
    )

    redator = Agent(
        role="Redator de Itiner√°rios",
        goal="Montar um resumo de viagem detalhado e atrativo com base nas informa√ß√µes coletadas",
        backstory="Profissional de turismo focado em apresentar planos de forma clara e encantadora.",
        verbose=True,
        llm=llm
    )

    tarefa1 = Task(
        description=f"Crie um esbo√ßo de viagem de {dias} dias para {destino}, com foco em: {interesses}.",
        agent=planejador
    )

    tarefa2 = Task(
        description=f"Com base no esbo√ßo, pesquise atra√ß√µes, pontos tur√≠sticos e dicas √∫teis sobre {destino}, focando em: {interesses}.",
        agent=pesquisador
    )

    tarefa3 = Task(
        description="Com base nas tarefas anteriores, crie um itiner√°rio final completo, claro e atrativo para o viajante.",
        agent=redator
    )

    equipe = Crew(
        agents=[planejador, pesquisador, redator],
        tasks=[tarefa1, tarefa2, tarefa3],
        verbose=True
    )

    resultado = equipe.kickoff()
    return resultado

# === Streamlit App ===
st.set_page_config(page_title="Planejador de Viagem", page_icon="üåç")
st.title("üåç Planejador de Viagem Personalizada")

st.markdown("Crie um roteiro de viagem sob medida com ajuda de agentes inteligentes! ‚úàÔ∏è")

with st.form("formulario_viagem"):
    destino = st.text_input("Destino")
    dias = st.number_input("N√∫mero de dias", min_value=1, step=1)
    interesses = st.text_input("Interesses (Ex: cultura, gastronomia, natureza)")
    submitted = st.form_submit_button("Gerar roteiro")

if submitted:
    with st.spinner("üß† Gerando plano de viagem..."):
        resultado = planejar_viagem(destino, dias, interesses)
        st.success("‚úÖ Roteiro gerado com sucesso!")

        tab1, tab2 = st.tabs(["üß≥ Itiner√°rio", "üó∫Ô∏è Mapa do destino"])

        with tab1:
            st.text_area("Seu roteiro:", resultado, height=400)
            st.download_button(
                label="üì• Baixar roteiro como .txt",
                data=resultado,
                file_name="itinerario_viagem.txt",
                mime="text/plain"
            )

        with tab2:
            lat, lon = obter_coordenadas(destino)
            if lat and lon:
                mapa = folium.Map(location=[lat, lon], zoom_start=12)
                folium.Marker([lat, lon], tooltip=destino).add_to(mapa)
                st_folium(mapa, width=700)
            else:
                st.warning("Localiza√ß√£o n√£o encontrada.")
